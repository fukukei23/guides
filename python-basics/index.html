ステップ2: リストと辞書

ステップ4: 繰り返し

# for文
for fruit in fruits:
    print(fruit)

# range
for i in range(5):
    print(i)  # 0,1,2,3,4

# while文
count = 0
while count < 5:
    count += 1

ステップ5: 関数

def greet(name):
    return f"Hello, {name}!"

result = greet("World")
print(result)  # Hello, World!

# デフォルト引数
def add(a, b=0):
    return a + b

ステップ7: クラス（オブジェクト指向）

class Dog:
    def __init__(self, name):
        self.name = name

    def bark(self):
        return f"{self.name} says woof!"

dog = Dog("Pochi")
print(dog.bark())  # Pochi says woof!
